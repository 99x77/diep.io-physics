$ Dm $

\begin{align} Dm_{A} = \begin{cases} \dfrac{3}{2} \cdot Df_{A} & E_{A} = E_{B} = Tank \\ \\ \dfrac{1}{4} \cdot Df_{A} & E_{B} = Bullet \\ \\ Df_{A} & otherwise \end{cases} \quad \quad \quad Dm_{B} = \begin{cases} \dfrac{3}{2} \cdot Df_{B} & E_{A} = E_{B} = Tank \\ \\ \dfrac{1}{4} \cdot Df_{B} & E_{A} = Bullet \\ \\ Df_{B} & otherwise \end{cases} \\ \end{align}

$ \Delta H $

\begin{align} \Delta H_{A} = \begin{cases} H_{A} & H_{A} - Dm_{B} \lt 0 \\ Dm_{B} & otherwise \end{cases} \\ \\ \Delta H_{B} = \begin{cases} H_{B} & H_{B} - Dm_{A} \lt 0 \\ Dm_{A} & otherwise \end{cases} \end{align}

$ Da $

\begin{align} Da_{A} &= Dm_{B} - \Delta H_{A}\biggl(1 - \dfrac{\Delta H_{B}}{Dm_{A}}\biggr) \\ \\ Da_{B} &= Dm_{A} - \Delta H_{B}\biggl(1 - \dfrac{\Delta H_{A}}{Dm_{B}}\biggr) \\ \end{align}
